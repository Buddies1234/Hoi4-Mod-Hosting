on_actions = {

	on_startup = {
		effect = {
			every_possible_country = { add_ideas = NO_DIV_hidden }
		}
	}

	on_weekly = {
		effect = {
			if = {
				limit = {
					NOT = { has_idea = NO_DIV_hidden }
				}
				add_ideas = NO_DIV_hidden
			}
		}
	}

	on_puppet = {
		effect = {
			if = {
				limit = {
					NOT = { has_idea = NO_DIV_hidden }
				}
				add_ideas = NO_DIV_hidden
			}
		}
	}
}
